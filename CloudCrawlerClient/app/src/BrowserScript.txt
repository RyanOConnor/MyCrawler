		 var x = document.getElementsByTagName('a');
		 for(var i = 0; i < x.length; i++){
			x[i].addEventListener("click", highlight, false);
		 }

		 function highlight() {
			this.style.backgroundColor = "yellow";
			this.style.color = "black";
			this.href = "javascript:void(0);";

			var node = this;
			var nodePath = node.tagName;
			if(node.className) {
				nodePath = nodePath + '.' + node.className;
			} else {
				while(!node.className) {
					node = node.parentNode;
					if(node.className){
						var fullName = '';
						if(node.className.indexOf(' ') != -1){
							var classes = node.className.split(' ');
							for(var i = 0; i < classes.length; i++){
								fullName = fullName + '.' + classes[i];
							}
						} else {
							fullName = '.' + node.className;
						}
						nodePath = node.tagName + fullName + ' ' + nodePath;
					} else {
						nodePath = node.tagName + ' ' + nodePath;
					}
				}
			}
			var nodes = document.querySelectorAll(nodePath);
			for(var i = 0; i < nodes.length; i++){
				this.style.backgroundColor = "yellow";
				this.style.color = "black";
				this.href = "javascript:void(0);";
			}

		 }
